ACTOR M60Clip : Ammo
{
   +INVENTORY.IGNORESKILL
   Inventory.MaxAmount 60
}

ACTOR M60 : Weapon
{
   inventory.pickupmessage "You got the M60!"
   Inventory.PickupSound "weapon/pickup"
   Weapon.AmmoType "M60Clip"
   Weapon.AmmoGive 0
   Weapon.AmmoUse 1
   Weapon.AmmoType2 "Clip"
   Weapon.AmmoGive2 50
   Weapon.AmmoUse2 0
   Weapon.Kickback 100
   Weapon.BobStyle InverseSmooth
   Weapon.BobSpeed 2.2
   Weapon.BobRangeY 0.3
   Weapon.BobRangeX 0.2
   Weapon.SelectionOrder 1303
   Weapon.SlotNumber 4
   AttackSound "m60/fire"
   Scale 0.6
   +NOAUTOAIM
   +NOAUTOFIRE
   +Weapon.Ammo_Optional 
   +INVENTORY.IGNORESKILL
   +weapon.noalert
   States
   {
   Ready:
     TNT1 A 0 A_JumpIFInventory("FireSel", 1, "AltAltFire")
     M60R A 1 A_WeaponReady(WRF_ALLOWRELOAD)
     Goto Ready
   Deselect:
	 TNT1 A 0 
     Goto DeselectAnim
    DeselectAnim:
	 TNT1 A 0 A_StartSound ("weapon/inventory")
	 M60R A 1
	 M6IN ABCDEF 1
	 Goto DeselectEnd
   DeselectEnd:
	 TNT1 AAAAAAAAAAAAAA 00 A_Lower
     Loop
   Select:
	 TNT1 A 1 
   	 TNT1 AAAAAAAAAAAAAA 00 A_Raise
     Goto SelectAnim	 
   SelectAnim:
	 TNT1 A 0 A_StartSound ("weapon/inventory")
	 M6IN FEDCBA 1
	 M60R A 1
	 TNT1 A 0 A_StartSound ("m60/ready")
	 Goto Ready
   AltAltFire:
	 TNT1 A 0 A_TakeInventory("FireSel",1)
 	 TNT1 A 0 A_StartSound ("weapon/firemode")
	 TNT1 A 0 A_JumpIFInventory("M60BurstFire", 0, "StopAlt")
	 TNT1 A 0 A_GiveInventory("M60BurstFire", 1)
	 TNT1 A 0 A_Print("Burst fire")
 	 Goto Ready		
   StopAlt:
	 TNT1 A 0 A_StartSound ("weapon/firemode")
	 TNT1 A 0 A_TakeInventory("M60BurstFire", 1)
	 TNT1 A 0 A_Print("Full Auto")
	 Goto Ready
   Fire:
	  TNT1 A 0 A_JumpIFInventory("M60BurstFire", 1, "Fire2")
      TNT1 A 0 A_JumpIfNoAmmo("NoAmmo")
      M60F A 1 A_FireBullets (3,4,-1,7,"BulletSpark")
	  TNT1 A 0 A_SetPitch(pitch-0.5)
	  TNT1 A 0 A_SetAngle(angle-0.2)
	  M60F B 1 
	  TNT1 A 0 BRIGHT A_GunFlash
	  TNT1 A 0 A_AlertMonsters
      TNT1 A 0 A_FireCustomMissile("RifleCasingSpawner",0,0,0,0)
	  TNT1 A 0 A_FireCustomMissile("MGLinksSpawner",0,0,0,0)
	  M60R A 1
	  TNT1 A 0 A_Refire
	  Goto Ready
    Fire2:
      TNT1 A 0 A_JumpIfNoAmmo("NoAmmo")
      M60F A 1 A_FireBullets (3,4,-1,7,"BulletSpark")
	  TNT1 A 0 A_SetPitch(pitch-0.5)
	  TNT1 A 0 A_SetAngle(angle-0.2)
	  M60F B 1 
	  TNT1 A 0 BRIGHT A_GunFlash
	  TNT1 A 0 A_AlertMonsters
      TNT1 A 0 A_FireCustomMissile("RifleCasingSpawner",0,0,0,0)
	  TNT1 A 0 A_FireCustomMissile("MGLinksSpawner",0,0,0,0)
	  M60R A 1
      TNT1 A 0 A_JumpIfNoAmmo("NoAmmo")
      M60F A 1 A_FireBullets (3,4,-1,7,"BulletSpark")
	  TNT1 A 0 A_SetPitch(pitch-0.5)
	  TNT1 A 0 A_SetAngle(angle-0.2)
	  M60F B 1 
	  TNT1 A 0 BRIGHT A_GunFlash
	  TNT1 A 0 A_AlertMonsters
      TNT1 A 0 A_FireCustomMissile("RifleCasingSpawner",0,0,0,0)
	  TNT1 A 0 A_FireCustomMissile("MGLinksSpawner",0,0,0,0)
	  M60R A 1
      TNT1 A 0 A_JumpIfNoAmmo("NoAmmo")
      M60F A 1 A_FireBullets (3,4,-1,7,"BulletSpark")
	  TNT1 A 0 A_SetPitch(pitch-0.5)
	  TNT1 A 0 A_SetAngle(angle-0.2)
	  M60F B 1 
	  TNT1 A 0 BRIGHT A_GunFlash
	  TNT1 A 0 A_AlertMonsters
      TNT1 A 0 A_FireCustomMissile("RifleCasingSpawner",0,0,0,0)
	  TNT1 A 0 A_FireCustomMissile("MGLinksSpawner",0,0,0,0)
	  M60R A 1
	  Goto Ready
	Reload:
	  TNT1 A 0 A_JumpIfInventory("M60Clip", 60, 2)
	  TNT1 A 0 A_JumpIfInventory("Clip", 1, "ReloadWork")
	  TNT1 A 0
	  Goto Ready
	ReloadWork:
	  TNT1 A 0 A_StartSound ("m60/reloadseq")
	  M6IN ABCDE 1
	  M6IN F 65
	  TNT1 A 0 A_StartSound ("m60/ready")
	  M6IN EDCBA 1
	  ReloadLoop:
	  TNT1 A 0 A_TakeInventory("Clip", 1)
	  TNT1 A 0 A_GiveInventory("M60Clip",1)
	  TNT1 A 0 A_JumpIfInventory("M60Clip", 60, "Full")
	  TNT1 A 0 A_JumpIfInventory("Clip", 1, "ReloadLoop")
	Goto Full
	Full:
      M60R A 0
    Goto Ready 
	NoAmmo:
	  TNT1 A 0 A_JumpIfInventory("M60Clip", 60, 2)
	  TNT1 A 0 A_JumpIfInventory("Clip", 1, "ReloadWork")
	  TNT1 A 0 A_StartSound ("weapon/noammo")
    Goto Ready 
	Flash:
      TNT1 A 4 Bright A_Light1
      TNT1 A 3 Bright A_Light2
    Goto LightDone
	Spawn:
	  M60I A -1
      Stop
	}
}
