ACTOR Cop2 Replaces DoomImp
{
  Health 80
  Radius 20
  Height 56
  Mass 100
  scale 0.32
  Speed 14
  PainChance 140
  Monster
  +FLOORCLIP
  +NOINFIGHTING
  SeeSound "cop/sight"
  AttackSound "cop/attack"
  PainSound "cop/pain"
  DeathSound "cop/death"
  Obituary "%o was shot by a cop."
  Dropitem "P50"
  States
  {
  Spawn:
    TNT1 A 0 A_TakeInventory("Cop2Clip", 10)
    TNT1 A 0 A_TakeInventory("Cop2ReloadCounter", 5)
    TNT1 A 0 A_GiveInventory("Cop2Clip", random(2,6))
    TNT1 A 0 A_GiveInventory("Cop2ReloadCounter", random(2,4))
    FCP1 A 10 A_Look
    Loop
  See:
    TNT1 A 0 A_JumpIfInventory("Cop2Clip", 10, "Reload")
    FCP1 AABBCCDD 3 A_Chase
    Loop
  Missile:
    TNT1 A 0 A_JumpIfInventory("Cop2ReloadCounter", 5, "Missile_NoAmmo")
    TNT1 A 0 A_GiveInventory("Cop2Clip", 1)
    FCP1 F 2 A_FaceTarget
	TNT1 A 0 A_CustomBulletAttack(13, 2, 1, random(1,5) * 3, "BulletSpark", 0, CBAF_NORANDOM)
	TNT1 A 0 A_SpawnItem ("PistolCasingSpawner", 0, 30,0)
	FCP1 F 5 bright
    FCP1 F 2
    Goto See
  Missile_NoAmmo:
    TNT1 A 0 A_TakeInventory("Cop2Clip", 10)
    FCP1 F 2 A_FaceTarget
	TNT1 A 0 A_PlaySound ("weapon/noammo")
    FCP1 F 5
    FCP1 F 2
    Goto See
  Reload:
    TNT1 A 0 A_GiveInventory("Cop2ReloadCounter", 1)
    TNT1 A 0 A_JumpIfInventory("Cop2ReloadCounter", 5, "Missile_NoAmmo")
	TNT1 A 0 A_TakeInventory("Cop2Clip", 10)
    TNT1 A 0 A_PlaySound("enemy/reload")
	TNT1 A 0 A_SpawnItem ("EmptyClipSpawner", 0,30,0)
    FCP1 E 5
	Goto See
  Pain:
    FCP1 E 3
    FCP1 E 3 A_Pain
    Goto See
 Death:
    FCP1 H 5
    FCP1 I 5 A_Scream
    FCP1 J 5 A_NoBlocking
    FCP1 K 5
    FCP1 L -1
    Stop
  XDeath:
    FCP1 H 5
    FCP1 I 5 A_Scream
    FCP1 J 5 A_NoBlocking
    FCP1 K 5
    FCP1 L -1
    Stop
  Raise:
    FCP1 L 5
    FCP1 KJIH 5
    Goto See
  }
}
