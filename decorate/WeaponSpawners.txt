ACTOR Glock18Spawner : CustomInventory Replaces Pistol 
{ 
   inventory.pickupmessage "You got the Glock 18!"
   Inventory.PickupSound "weapon/pickup"
   Scale 0.2
   States 
   { 
   Spawn: 
   IGLK A -1
   Stop
   Pickup:
   TNT1 A 0 A_GiveInventory("Glock18", 1)
   Goto Spawn
   TNT1 A 0
   Fail
   }
}
